#include <stdio.h>
#include <string.h>

//Khai báo struct
struct Student
{
int ID;
char Name[50];
char Email[20];
float Score;
};

//Khai báo các hàm nguyên mẫu (prototype)
void InputStudent(Student stds[], int &number);
void OutputStudent(Student stds[], int number);

void main()
{
int Number;
Student Class[20];
InputStudent(Class, Number);
OutputStudent(Class, Number);
}

//Xây dựng hàm nhập học viên
void InputStudent(Student stds[], int &number)
{
printf(“Moi nhap so luong HV: ”);
scanf(“%d”,&number);
for(int i=0; i<number; i++)
{
printf(“Nhap sinh vien thu %d\n”, i+1);
printf(“Ma so: ”); scanf(“%d”, &stds[i].ID);
fflush(stdin); //Lệnh xoá bộ nhớ đệm, tránh tình trạng lệnh bị trôi.
printf(“Ho ten: ”); gets(stds[i].Name);
printf(“Email: ”); gets(stds[i].Email);
printf(“Score: ”); scanf(“%f”,&stds[i].Score);
}
}
//Xây dựng hàm xuất danh sách học viên
void OutputStudent(Student stds[], int number)
{
for(int i=0; i<number; i++)
{
printf(“MSHV: %d\nTen: %s\n”, stds[i].ID, stds[i].Name);
printf(“Email: %s\nScore: %.1f\n”, stds[i].Email, stds[i].Score);
}
}
